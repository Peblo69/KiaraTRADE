Approach for Full Bulgarian Language Integration
Detect the user's language dynamically (based on input).
Use pre-defined Bulgarian phrases for common responses or full conversations.

Make sure her responses match Bulgarian syntax, grammar, and tone, so they feel natural.
Step 1: Language Detection
Implement a simple function to detect if the user is speaking in Bulgarian. For this, you can use language detection libraries like franc or build your own detection based on character sets and common words.

Example Language Detection Function
typescript
Copy
Edit
import franc from "franc";  // Install franc for language detection

function detectLanguage(userMessage: string): string {
  const detectedLang = franc(userMessage, { minLength: 3 });
  if (detectedLang === "bul") {
    return "bg";  // Bulgarian detected
  }
  return "en";  // Default to English
}

const languagePacks = {
  en: {
    greeting: "Hey there! How can I help you today?",
    masterMode: "Yes, Master. What would you like me to do?",
  },
  bg: {
    greeting: "Здравейте! Как мога да ви помогна днес?",
    masterMode: "Да, Господарю. Какво бихте желали да направя?",
  },
};

function getResponse(key: string, language: string): string {
  return languagePacks[language]?.[key] || languagePacks["en"][key];
}

export function generateResponse(userMessage: string): string {
  const language = detectLanguage(userMessage);
  const currentMode = isCreatorModeActive() ? "masterMode" : "greeting";

  return getResponse(currentMode, language);
}
